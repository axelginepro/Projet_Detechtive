! ════════════════════════════════════════
! ACLs DE SEGMENTATION INTER-VLAN
! [MODE LAB] ICMP autorisé partout pour tests de connectivité
! Appliquées en entrée (in) sur chaque SVI
! ════════════════════════════════════════

! ────────────────────────────────────────
! DMZ (VLAN 10) — 192.168.10.0/28
! ────────────────────────────────────────
ip access-list extended ACL_DMZ_IN
 remark [LAB] ICMP autorisé depuis DMZ vers tous les VLANs (ping de test)
 permit icmp 192.168.10.0 0.0.0.15 any
 remark Autoriser Web (192.168.10.10) -> MariaDB (192.168.10.21) port MySQL
 permit tcp host 192.168.10.10 host 192.168.10.21 eq 3306
 remark Autoriser DMZ -> Wazuh Manager (agent ports 1514/1515)
 permit tcp 192.168.10.0 0.0.0.15 host 192.168.10.42 eq 1514
 permit tcp 192.168.10.0 0.0.0.15 host 192.168.10.42 eq 1515
 remark Bloquer DMZ -> Serveurs internes [LOG -> Wazuh Syslog]
 deny   ip 192.168.10.0 0.0.0.15 192.168.10.16 0.0.0.15 log
 remark Bloquer DMZ -> Postes clients [LOG -> Wazuh Syslog]
 deny   ip 192.168.10.0 0.0.0.15 192.168.10.128 0.0.0.127 log
 remark Bloquer DMZ -> Active Directory [LOG -> Wazuh Syslog]
 deny   ip 192.168.10.0 0.0.0.15 192.168.10.32 0.0.0.7 log
 remark Bloquer DMZ -> Reste Management (hors Wazuh) [LOG -> Wazuh Syslog]
 deny   ip 192.168.10.0 0.0.0.15 192.168.10.40 0.0.0.7 log
 remark Autoriser DMZ -> Internet via pfSense
 permit ip 192.168.10.0 0.0.0.15 any
exit

! ────────────────────────────────────────
! SERVEURS INTERNES (VLAN 20) — 192.168.10.16/28
! ────────────────────────────────────────

 remark [LAB] ICMP autorisé depuis VLAN 20 vers tous les VLANs (ping de test)
 permit icmp 192.168.10.16 0.0.0.15 any
 remark Autoriser Serveurs -> DMZ : maintenance HTTP/HTTPS
 permit tcp 192.168.10.16 0.0.0.15 host 192.168.10.10 eq 80
 permit tcp 192.168.10.16 0.0.0.15 host 192.168.10.10 eq 443
 remark Autoriser Serveurs -> AD : DNS, LDAP, Kerberos
 permit udp 192.168.10.16 0.0.0.15 192.168.10.32 0.0.0.7 eq 53
 permit tcp 192.168.10.16 0.0.0.15 192.168.10.32 0.0.0.7 eq 389
 permit tcp 192.168.10.16 0.0.0.15 192.168.10.32 0.0.0.7 eq 88
 remark Autoriser Serveurs -> Wazuh Manager (agent ports 1514/1515)
 permit tcp 192.168.10.16 0.0.0.15 host 192.168.10.42 eq 1514
 permit tcp 192.168.10.16 0.0.0.15 host 192.168.10.42 eq 1515
 remark Autoriser Serveurs -> Admin réseau (192.168.10.43) : SSH/gestion
 permit tcp 192.168.10.16 0.0.0.15 host 192.168.10.43 eq 22
 remark Bloquer Serveurs -> Postes clients [LOG -> Wazuh Syslog]
 deny   ip 192.168.10.16 0.0.0.15 192.168.10.128 0.0.0.127 log
 remark Autoriser Serveurs -> Internet (mises à jour, NTP...)
